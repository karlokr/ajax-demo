CREATE TABLE reviews (
ID int NOT NULL AUTO_INCREMENT,
name VARCHAR(50),
date_written VARCHAR(20),
location VARCHAR(30),
comment VARCHAR(300),
number_of_stars INT NOT NULL,
PRIMARY KEY (ID)
);

CREATE TABLE facilities (
ID int NOT NULL AUTO_INCREMENT,
name varchar(30),
address VARCHAR(50),
city VARCHAR(20),
postal_code VARCHAR(6),
map VARCHAR(500),
PRIMARY KEY (ID)
);
